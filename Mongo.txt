db.database1_table1.find()

db.database1_table1.find({x:1,z:3})

// the same. use "" or not.
db.database1_table1.insert({"x":1})
db.database1_table1.insert({x:1})
db.database1_table1.insert({x:1,y:2,z:3})

// delete document
db.database1_table1.remove({x:2})
